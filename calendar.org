def content_to_string content_type, content
  if content_type == "Q&A"
     "Q&A on lessons #{content}."
  else
     content
  end
end

def link_if file, name
  File.exists?(file) ? "<a href=\"#{file}\">#{name}</a>" : name
end

def content_to_material content, material
  mat = material.select { |x| (content || "").split(",").include?(x["topic_no"]) }.map { |x| x["basename"] }
  mat.map { |x|
    video = "./material/#{x}-small.mp4";
    pdf   = "./material/#{x}.pdf";
    "#{x} (#{link_if(video, "VIDEO")}) (#{link_if(pdf, "PDF")})"
  }.join("<br/>\n")
end

    <td class="material">

    </td>
    <td class="content">
      <%= content_to_string slot["content_type"], slot["content"] %>

    </tr>
    <% end %>
  </tbody>
</table>
EOS

  array.each do |row|
